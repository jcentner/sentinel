Recommended Configurations

Privacy-first (all local)

[sentinel]
provider = "ollama"
model = "qwen3.5:4b"
model_capability = "basic"
# Skip test-coherence — 4B model has ~40% FP rate
disabled_detectors = ["test-coherence"]

Best quality (hybrid local + cloud)

[sentinel]
provider = "ollama"
model = "qwen3.5:4b"
model_capability = "basic"

# Route test-coherence to cloud for better precision
[sentinel.detector_providers.test-coherence]
provider = "openai"
model = "gpt-5.4-nano"
api_base = "https://api.openai.com/v1"
api_key_env = "OPENAI_API_KEY"
model_capability = "standard"

No LLM (fastest, deterministic only)

[sentinel]
skip_judge = true
skip_llm = true
# Runs all heuristic + deterministic detectors with no model at all
